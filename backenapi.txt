## Quick Reference

### Base URL
```
http://localhost:8000  (Development)
https://api.fashionai.com  (Production - configure in .env)
```

### Required Headers (All Authenticated Endpoints)
```
Authorization: Bearer <access_token>
Content-Type: application/json (or multipart/form-data for file uploads)
```

### Common Error Responses
```json
{
  "detail": "Error message here"
}
```

| Status Code | Meaning |
|---|---|
| 200 | Success |
| 201 | Created |
| 204 | No Content (successful delete) |
| 400 | Bad Request (validation error) |
| 401 | Unauthorized (invalid/missing token) |
| 403 | Forbidden (insufficient permissions) |
| 404 | Not Found |
| 500 | Server Error |

```env
# Backend API Configuration
VITE_API_BASE_URL=http://localhost:8000
VITE_API_TIMEOUT=30000

# Razorpay Payment Gateway
VITE_RAZORPAY_KEY_ID=your_razorpay_key_id

# Feature Flags
VITE_ENABLE_TRY_ON=true
VITE_ENABLE_RECOMMENDATIONS=true
VITE_ENABLE_PAYMENTS=true

# Image Upload
VITE_MAX_IMAGE_SIZE=10485760  # 10MB in bytes
VITE_ALLOWED_IMAGE_TYPES=image/jpeg,image/png,image/webp

# Try-On Processing
VITE_TRYON_POLLING_INTERVAL=2000  # Poll every 2 seconds
VITE_TRYON_MAX_POLLING_TIME=300000  # Stop after 5 minutes

# Storage
VITE_MEDIA_BASE_URL=http://localhost:8000/media
```

### Initialize API Client (Frontend Setup)
```typescript
// services/api.ts
import axios from 'axios';

const API = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || 'http://localhost:8000',
  timeout: parseInt(import.meta.env.VITE_API_TIMEOUT || '30000'),
});

// Request interceptor
API.interceptors.request.use((config) => {
  const token = localStorage.getItem('access_token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Response interceptor for token refresh
API.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('access_token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default API;
```

# Authentication Module

## 1. Registration Step 1 (Email Verification + Account Creation)

### Page Name
**Registration Page - Step 1 (Email Verification)**

### Purpose
Users submit email and password; system sends OTP code to verify email ownership before creating account.

### APIs Used on This Page
1. `POST /auth/send-otp` - Send OTP to email
2. `POST /auth/verify-otp` - Verify OTP code
3. `POST /auth/register/step-1` - Create account after email verification

### API 1: Send OTP

**HTTP Method:** `POST`

**Endpoint URL:** `POST /auth/send-otp`

**Authentication Required:** No

**Required Headers:** 
```
Content-Type: application/json
```

**Request Body:**
```json
{
  "email": "user@example.com"
}
```

**Response JSON Example (Success 200):**
```json
{
  "success": true,
  "message": "OTP sent to your email"
}
```

**Response JSON Example (Failure 400):**
```json
{
  "success": false,
  "message": "Email sending failed"
}
```

**Validation Rules:**
- Email must be valid email format
- Email must not already be registered

**Error Responses:**
| Code | Error | Solution |
|---|---|---|
| 400 | Invalid email format | Use valid email |
| 400 | Email already registered | Use different email or login |
| 500 | Email service error | Retry after few seconds |

### API 2: Verify OTP

**HTTP Method:** `POST`

**Endpoint URL:** `POST /auth/verify-otp`

**Authentication Required:** No

**Required Headers:**
```
Content-Type: application/json
```

**Request Body:**
```json
{
  "email": "user@example.com",
  "otp_code": "123456"
}
```

**Response JSON Example (Success 200):**
```json
{
  "success": true,
  "message": "Email verified successfully"
}
```

**Response JSON Example (Failure 400):**
```json
{
  "success": false,
  "message": "Invalid OTP code"
}
```

**Validation Rules:**
- OTP code must be exactly 6 digits
- OTP must be numeric only
- OTP valid for 10 minutes after sending

**Error Responses:**
| Code | Error | Solution |
|---|---|---|
| 400 | Invalid OTP | Enter correct code sent to email |
| 400 | OTP expired | Request new OTP |
| 400 | Email not found | Send OTP first |

**User Workflow:**
1. User enters email, clicks "Send OTP"
2. Backend sends OTP to email → Show "Check your email" message

**File Upload Requirements:**
```typescript
const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10MB
const ALLOWED_TYPES = ['image/jpeg', 'image/png', 'image/webp'];

const validateFile = (file: File): { valid: boolean; error?: string } => {
  if (!file) return { valid: false, error: 'No file selected' };
  if (!ALLOWED_TYPES.includes(file.type)) {
    return { valid: false, error: 'Only JPEG, PNG, WebP allowed' };
  }
  if (file.size > MAX_FILE_SIZE) {
    return { valid: false, error: 'File must be under 10MB' };
  }
  return { valid: true };
};
```

```typescript
import API from '@/services/api';

const handlePhotoSelect = (e: React.ChangeEvent<HTMLInputElement>) => {
  const file = e.target.files?.[0];
  if (!file) return;
  
  const validation = validateFile(file);
  if (!validation.valid) {
    setError(validation.error || 'Invalid file');
    return;
  }
  
  setProfilePhoto(file);
  const reader = new FileReader();
  reader.onloadend = () => setPreviewUrl(reader.result as string);
  reader.readAsDataURL(file);
  setError('');

const handleSubmit = async () => {
  if (!genderPreference || !profilePhoto) {
    setError('Please fill all fields');
    return;
  }
  
  setLoading(true);
  try {
    const formData = new FormData();
    formData.append('gender_preference', genderPreference);
    formData.append('photo', profilePhoto);
    
    const response = await API.post('/auth/register/step-2', formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
      onUploadProgress: (progressEvent) => {
        const percentCompleted = Math.round(
          (progressEvent.loaded * 100) / (progressEvent.total || 1)
        );
        setUploadProgress(percentCompleted);
      },
    });
    
    localStorage.setItem('user', JSON.stringify(response.data));
    navigate('/home');
  } catch (err) {
    setError(err.response?.data?.detail || 'Registration failed');
  } finally {
    setLoading(false);
    setUploadProgress(0);
  }
};
```

```typescript
const handleLogin = async () => {
  setLoading(true);
  try {
    const response = await API.post('/auth/login', {
      email,
      password,
    });
    
    // Store token and user data
    localStorage.setItem('access_token', response.data.access_token);
    localStorage.setItem('user', JSON.stringify(response.data.user));
    localStorage.setItem('token_type', response.data.token_type);
    localStorage.setItem('kind', response.data.kind);

      navigate('/register/step-2');

  } catch (err) {
    setError(err.response?.data?.detail || 'Login failed');

    setLoading(false);

};

const handleLogout = () => {

  localStorage.removeItem('user');
  localStorage.removeItem('token_type');
  localStorage.removeItem('kind');
  navigate('/login');
};
```

```typescript
const createTryOn = async (topItemId: string, overridePhoto?: File) => {
  const formData = new FormData();
  formData.append('top_item_id', topItemId);
  if (overridePhoto) {
    formData.append('override_photo', overridePhoto);
  }
  formData.append('garment_photo_type', 'flat-lay');
  
  try {
    const response = await API.post('/tryon', formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
    });
    
    return response.data;
  } catch (err) {
    if (err.response?.status === 400) {
      throw new Error(err.response.data.detail);
    }
    throw err;
  }
};

// Polling mechanism for job completion
const pollTryOnStatus = async (jobId: string, maxWait = 300000) => {
  const pollInterval = 2000; // 2 seconds
  const startTime = Date.now();
  
  while (Date.now() - startTime < maxWait) {
    try {
      const response = await API.get(`/tryon/history`);
      const job = response.data.find(j => j.id === jobId);
      
      if (job?.status === 'completed') {
        return job;
      } else if (job?.status === 'failed') {
        throw new Error(job.error_message);
      }
    } catch (err) {
      console.error('Polling error:', err);
    }
    
    await new Promise(resolve => setTimeout(resolve, pollInterval));
  }
  
  throw new Error('Try-on processing timeout');
};
```

```typescript
API.interceptors.response.use(
  (response) => response,

    const originalRequest = error.config;
```

```typescript
// Generic file upload handler
const uploadFile = async (
  endpoint: string,
  formFields: Record<string, string>,
  file: File,
  onProgress?: (percent: number) => void
) => {
  const formData = new FormData();
  
  // Add form fields
  Object.entries(formFields).forEach(([key, value]) => {
    formData.append(key, value);
  });
  
  // Add file
  formData.append('file', file);
  
  try {
    const response = await API.post(endpoint, formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
      onUploadProgress: (progressEvent) => {
        const percentCompleted = Math.round(
          (progressEvent.loaded * 100) / (progressEvent.total || 1)
        );
        onProgress?.(percentCompleted);
      },
    });
    
    return response.data;
  } catch (err) {
    throw new Error(err.response?.data?.detail || 'Upload failed');
  }
};
```

## AI Processing Flow

```typescript
// Try-on processing with polling
const processTryOn = async (topItemId: string, maxWait = 300000) => {
  const pollInterval = 2000; // 2 seconds
  const maxAttempts = Math.ceil(maxWait / pollInterval);
  let attempts = 0;
  
  // Create try-on job
  const jobResponse = await API.post('/tryon', new FormData(), {
    top_item_id: topItemId,
  });
  
  const jobId = jobResponse.data.id;
  let status = jobResponse.data.status;
  
  // Poll for completion
  while (status === 'processing' && attempts < maxAttempts) {
    await new Promise(resolve => setTimeout(resolve, pollInterval));
    
    const historyResponse = await API.get('/tryon/history');
    const job = historyResponse.data.find(j => j.id === jobId);
    
    status = job?.status;
    attempts++;
    
    if (status === 'completed') {
      return job;
    } else if (status === 'failed') {
      throw new Error(job?.error_message || 'Processing failed');
    }
  }
  
  throw new Error('Processing timeout');
};
```

```typescript
            <Route path="checkout/:planCode" element={<CheckoutPage />} />
            <Route path="settings" element={<SettingsPage />} />
          </Route>
        </Route>
        
        {/* Admin Routes */}
        <Route element={<AdminRoute />}>
          <Route path="/admin" element={<AdminLayout />}>
            <Route index element={<AdminDashboard />} />
            <Route path="plans" element={<AdminPlansPage />} />
            <Route path="users" element={<AdminUsersPage />} />
            <Route path="analytics" element={<AdminAnalyticsPage />} />
          </Route>
        </Route>
        
        {/* Error Routes */}
        <Route path="*" element={<NotFoundPage />} />
      </Routes>
    </BrowserRouter>
  );
}

// ProtectedRoute.tsx
interface ProtectedRouteProps {
  requiredFull?: boolean; // Require full registration
}

const ProtectedRoute: React.FC<ProtectedRouteProps> = ({ requiredFull = true }) => {
  const { isAuthenticated, user } = useAuth();
  
  if (!isAuthenticated) {
    return <Navigate to="/login" replace />;
  }
  
  if (requiredFull && !user?.is_fully_registered) {
    return <Navigate to="/register/step-2" replace />;
  }
  
  return <Outlet />;
};
```

## Error Handling Strategy

```typescript
// utils/errorHandler.ts
interface APIError {
  status: number;
  message: string;
  detail?: string;
  field?: string;
}

class ErrorHandler {
  static handle(error: any): APIError {
    if (error.response?.status === 401) {
      return {
        status: 401,
        message: 'Session expired. Please login again.',
        detail: 'Unauthorized',
      };
    }
    
    if (error.response?.status === 403) {
      return {
        status: 403,
        message: 'You do not have permission to access this resource.',
        detail: error.response.data?.detail,
      };
    }
    
    if (error.response?.status === 404) {
      return {
        status: 404,
        message: 'Resource not found.',
        detail: error.response.data?.detail,
      };
    }
    
    if (error.response?.status === 400) {
      return {
        status: 400,
        message: 'Invalid request.',
        detail: error.response.data?.detail,
      };
    }
    
    if (error.response?.status === 500) {
      return {
        status: 500,
        message: 'Server error. Please try again later.',
        detail: error.response.data?.detail,
      };
    }
    
    return {
      status: error.response?.status || 0,
      message: error.message || 'An error occurred.',
      detail: error.response?.data?.detail,
    };
  }
}

// Usage
try {
  await API.post('/tryon', formData);
} catch (err) {
  const apiError = ErrorHandler.handle(err);
  showNotification(apiError.message, 'error');
  console.error('API Error:', apiError);
}
```
